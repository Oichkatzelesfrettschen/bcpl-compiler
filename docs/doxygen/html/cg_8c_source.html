336static void buf_append(char **sp, size_t *rem, const char *fmt, ...)
337{
338 int n;
339 va_list ap;
340
341 if (*rem == 0) {
342 error("instruction too long");
343 }
344 va_start(ap, fmt);
345 n = vsnprintf(*sp, *rem, fmt, ap);
346 va_end(ap);
347 if (n < 0 || (size_t)n >= *rem) {
348 error("instruction too long");
349 }
350 *sp += n;
351 *rem -= n;
352}

368static ocode_op gencode(void)
369{
370 bool ro;
371 int rcode, s1, s2, s3, sn;
372 ocode_op op;
373
374 dt = sp = lp = rcode = 0;
375 ro = false;
376 emit(".text");
377 for (;;)
378 {
379 op = rdop(0);
380 if (ch == EOF) {
381 return S_END;
382 }
383 if (op < 0 || op > OPMAX) {
384 error("Bad op %d", op);
385 }
386 s1 = optab1(op);
387 s2 = optab2(op);
388 s3 = optab3(op);
389 if (s3 <= 7) {
390 force(s1);
391 if (s3) {
392 loadreg(0, s3 == 1 ? 0 : s3 == 2 ? ltype[0] != X_N : 1);
393 }
394 if (s3 >= 4) {
395 loadreg(1, s3 == 4 ? 0 : s3 == 5 ? ltype[1] == X_N : 1);
396 }
397 }
398 switch (op) {
399 case S_LN:
400 load(X_N, rdn());
401 break;
402 case S_TRUE:
403 load(X_N, -1);
404 break;
405 case S_FALSE:
406 load(X_N, 0);
407 break;
408 case S_LP:
409 load(X_P, rdn());
410 break;
411 case S_LG:
412 load(X_G, rdn());
413 break;
414 case S_LL:
415 load(X_L, rdn());
416 break;
417 case S_LLP:
418 load(X_LP, rdn());
419 break;
420 case S_LLG:
421 load(X_LG, rdn());
422 break;
423 case S_LLL:
424 load(X_LL, rdn());
425 break;
426 case S_QUERY:
427 load(X_R, lp != 0);
428 break;
429 case S_LSTR:
430 {
431 int n, l, i;
432 n = rdn();
433 l = --labno;
434 defdata(S_DATALAB, l);
435 defdata(S_LSTR, n);
436 for (i = 1; i <= n; i++) {
437 defdata(S_LSTR, rdn());
438 }
439 load(X_LL, l);
440 }
441 break;
442 case S_SP:
443 save(X_P, rdn());
444 break;
445 case S_SG:
446 save(X_G, rdn());
447 break;
448 case S_SL:
449 save(X_L, rdn());
450 break;
451 case S_ENTRY:
452 {
453 int n, l, i;
454 n = rdn();
455 l = rdn();
456 printf("//\t");
457 for (i = 0; i < n; i++) {
458 putchar(rdn());
459 }
460 putchar('\n');
461 codelab(l);
462#if BITS==64
463 emit("pop (%%rcx)");
464 emit("mov %%rbp,%d(%%rcx)", WORDSZ);
465 emit("mov %%rcx,%%rbp");
466#else
467 emit("pop (%%ecx)");
468 emit("mov %%ebp,%d(%%ecx)", WORDSZ);
469 emit("mov %%ecx,%%ebp");
470#endif
471 }
472 break;
473 case S_ENDPROC:
474 rdn();
475 break;
476 case S_SAVE:
477 case S_STACK:
478 sn = rdn();
479 break;
480 case S_STORE:
481 break;
482 case S_RV:
483#if BITS==64
484 emit("mov (,%%rax,%d),%%eax", WORDSZ);
485#else
486 emit("mov (,%%eax,%d),%%eax", WORDSZ);
487#endif
488 break;
489 case S_ABS:
490 emit("test %%eax,%%eax");
491 emit("jns 1f");
492 emit("neg %%eax");
493 emit("1:");
494 break;
495 case S_NEG:
496 case S_NOT:
497 codex(op == S_NEG ? X_NEG : X_NOT);
498 break;
499 case S_MULT:
500 case S_DIV:
501 case S_REM:
502 if (op == S_DIV || op == S_REM) {
503#if BITS==64
504 emit("cqto");
505#else
506 emit("cltd");
507#endif
508 }
509 codex(op == S_MULT ? X_IMUL : X_IDIV);
510 if (op == S_REM) {
511 emit("mov %%edx,%%eax");
512 }
513 break;
514 case S_PLUS:
515 case S_MINUS:
516 codex(op == S_MINUS ? X_SUB : X_ADD);
517 break;
518 case S_EQ:
519 case S_NE:
520 case S_LS:
521 case S_GR:
522 case S_LE:
523 case S_GE:
524 {
525 int o2;
526 codex(X_CMP);
527 o2 = rdop(1);
528 if (o2 == S_JT || o2 == S_JF) {
529 ro = true;
530 rcode = op;
531 } else {
532 emit("set%s %%al", relstr[op - S_EQ][1]);
533 emit("movzx %%al,%%eax");
534 emit("neg %%eax");
535 }
536 }
537 break;
538 case S_LSHIFT:
539 case S_RSHIFT:
540 emit("sh%cl %%cl,%%eax", op == S_RSHIFT ? 'r' : 'l');
541 break;
542 case S_LOGAND:
543 case S_LOGOR:
544 codex(op == S_LOGOR ? X_OR : X_AND);
545 break;
546 case S_EQV:
547 case S_NEQV:
548 if (op == S_EQV) {
549 emit("xorl $-1,%%eax");
550 }
551 codex(X_XOR);
552 break;
553 case S_GETBYTE:
554 case S_PUTBYTE:
555 emit("shl $%d,%%eax", WORD_SHIFT);
556 codex(X_ADD);
557 if (op == S_GETBYTE) {
558#if BITS==64
559 emit("movzb (%%rax),%%eax");
560#else
561 emit("movzb (%%eax),%%eax");
562#endif
563 } else {
564 code(X_MOV, X_R, 1, X_P, sp - 3);
565#if BITS==64
566 emit("mov %%cl,(%%rax)");
567#else
568 emit("mov %%cl,(%%eax)");
569#endif
570 sp--;
571 }
572 break;
573 case S_STIND:
574#if BITS==64
575 emit("mov %%eax,(,%%rcx,%d)", WORDSZ);
576#else
577 emit("mov %%eax,(,%%ecx,%d)", WORDSZ);
578#endif
579 break;
580 case S_GOTO:
581 codex(X_JMP);
582 break;
583 case S_JT:
584 case S_JF:
585 if (ro) {
586 emit("j%s %s", relstr[rcode - S_EQ][op == S_JT], label(rdn()));
587 ro = false;
588 } else {
589 emit("orl %%eax,%%eax");
590 emit("j%s %s", op == S_JF ? "z" : "nz", label(rdn()));
591 }
592 break;
593 case S_JUMP:
594 emit("jmp %s", label(rdn()));
595 break;
596 case S_SWITCHON:
597 {
598 int n, d, l, i;
599 n = rdn();
600 d = rdn();
601 l = --labno;
602 defdata(S_DATALAB, l);
603 for (i = 0; i < n; i++) {
604 defdata(S_ITEMN, rdn());
605 defdata(S_ITEML, rdn());
606 }
607#if BITS==64
608 emit("mov $%s,%%rdx", label(l));
609 emit("mov $%d,%%rcx", n);
610#else
611 emit("mov $%s,%%edx", label(l));
612 emit("mov $%d,%%ecx", n);
613#endif
614 emit("jecxz 2f");
615#if BITS==64
616 emit("1:cmp (%%rdx),%%eax");
617#else
618 emit("1:cmp (%%edx),%%eax");
619#endif
620 emit("je 3f");
621#if BITS==64
622 emit("add $8,%%rdx");
623#else
624 emit("add $8,%%edx");
625#endif
626 emit("loop 1b");
627 emit("2:jmp %s", label(d));
628#if BITS==64
629 emit("3:jmp *%d(%%rdx)", WORDSZ);
630#else
631 emit("3:jmp *%d(%%edx)", WORDSZ);
632#endif
633 }
634 break;
635 case S_RES:
636 emit("jmp %s", label(rdn()));
637 break;
638 case S_RSTACK:
639 sn = rdn();
640 ltype[0] = X_R;
641 ldata[0] = 0;
642 break;
643 case S_FNAP:
644 case S_RTAP:
645 sn = rdn();
646 code(X_LEA, X_R, 1, X_P, sn);
647 codex(X_CALL);
648 if (op == S_FNAP) {
649 ltype[0] = X_R;
650 ldata[0] = 0;
651 }
652 break;
653 case S_FNRN:
654 case S_RTRN:
655#if BITS==64
656 emit("mov %%rbp,%%rcx");
657 emit("mov %d(%%rcx),%%eax", WORDSZ);
658 emit("mov %%eax,%%ebp");
659 emit("mov (%%rcx),%%eax");
660 emit("jmp *%%rax");
661#else
662 emit("mov %%ebp,%%ecx");
663 emit("mov %d(%%ecx),%%ebp", WORDSZ);
664 emit("jmp *(%%ecx)");
665#endif
666 break;
667 case S_ENDFOR:
668 codex(X_CMP);
669 emit("jle %s", label(rdn()));
670 break;
671 case S_BLAB:
672 case S_LAB:
673 codelab(rdn());
674 break;
675 case S_DATALAB:
676 case S_ITEML:
677 case S_ITEMN:
678 defdata(op, rdn());
679 break;
680 case S_NEEDS:
681 case S_SECTION:
682 {
683 int n, i;
684 n = rdn();
685 printf("//\t%s: ", op == S_NEEDS ? "NEEDS" : "SECTION");
686 for (i = 0; i < n; i++) {
687 putchar(rdn());
688 }
689 putchar('\n');
690 }
691 break;
692 case S_GLOBAL:
693 {
694 int n, x, i;
695 emit("ret");
696 emit(".data");
697 for (i = 0; i < dt; i++) {
698 outdata(sdata[i][0], sdata[i][1]);
699 }
700 n = rdn();
701 for (i = 0; i < n; i++) {
702 x = rdn();
703 emit(".global G%d", x);
704 emit(".equ G%d,%s", x, label(rdn()));
705 }
706 /* end of code stream: return for next phase */
707 return op;
708 }
709 /* not reached */
710 case S_FINISH:
711 emit("jmp finish");
712 break;
713 default:
714 error("Unknown op %d", op);
715 }
716 /* adjust stack pointer */
717 sp = s2 & 2 ? sn : sp - s1;
718 if (s2 & 1)
719 sp++;
720 /* adjust load stack pointer */
721 if (s3 <= 7) {
722 lp = s2 & 1;
723 } else if (s3 == 8 && lp < 2) {
724 lp++;
725 }
726 }
727 return op;
728}

745static void force(int st)
746{
747 assert(lp >= 0 && lp <= 2);
748 assert(st >= 0 && st <= 2);
749 if (lp == st) {
750 return;
751 }
752 if (lp == 0) {
753 ltype[0] = X_R;
754 ldata[0] = 0;
755 code(X_MOV, ltype[0], ldata[0], X_P, sp - 1);
756 lp = 1;
757 if (lp == st) {
758 return;
759 }
760 }
761 assert(lp != 0);
762 if (st == 2) {
763 assert(ltype[0] == X_R && ldata[0] == 0);
764 emit("mov %%eax,%%ecx");
765 code(X_MOV, ltype[0], ldata[0], X_P, sp - 2);
766 ltype[1] = X_R;
767 ldata[1] = 1;
768 lp = 2;
769 } else {
770 do
771 {
772 loadreg(0, ltype[0] != X_N);
773 code(X_MOV, X_P, sp - lp, ltype[0], ldata[0]);
774 if (lp == 2) {
775 ltype[0] = ltype[1];
776 ldata[0] = ltype[1] == X_R ? 0 : ldata[1];
777 }
778 } while (--lp != st);
779 }
780}

782static void loadreg(int i, int must)
783{
784 int t, p;
785
786 t = ltype[i];
787 if (t == X_R) {
788 return;
789 }
790 p = t == X_LP || t == X_LG || t == X_LL;
791 if (p || must) {
792 code(p && t != X_LL ? X_LEA : X_MOV, X_R, i, t, ldata[i]);
793 if (p) {
794 emit("shr $%d,%%%s", WORD_SHIFT, reg[i]);
795 }
796 ltype[i] = X_R;
797 ldata[i] = i;
798 }
799}

806static void code(int xi, ...)
807{
808 /*
809 * A 64-byte buffer easily accommodates the longest formatted
810 * instruction which is well under 32 characters.
811 */
812 char buf[64], *s;
813 size_t rem = sizeof(buf);
814 int typ[2], dat[2];
815 va_list ap;
816 int cj, i1, na, x, i, t, d;
817
818 va_start(ap, xi);
819 x = xitab[xi];
820 cj = x & XCJ;
821 i1 = x & XI1 ? 1 : 0;
822 na = x & XNA;
823 for (i = 0; i < na; i++)
824 {
825 typ[i] = va_arg(ap, int);
826 dat[i] = va_arg(ap, int);
827 }
828 s = buf;
829 buf_append(&s, &rem, "%s", xistr[xi]);
830 for (i = na - 1; i >= i1; i--)
831 {
832 if (rem <= 1) {
833 error("instruction too long");
834 }
835 *s++ = i == na - 1 ? ' ' : ',';
836 rem--;
837 t = typ[i];
838 d = dat[i];
839 if (t <= 3) {
840 if (cj) {
841 if (rem <= 1) {
842 error("instruction too long");
843 }
844 *s++ = '*';
845 rem--;
846 }
847 } else if (t != X_LP && t != X_LG && !cj) {
848 if (rem <= 1) {
849 error("instruction too long");
850 }
851 *s++ = '$';
852 rem--;
853 }
854 switch (t)
855 {
856 case X_R:
857 buf_append(&s, &rem, "%%%s", reg[d]);
858 break;
859 case X_P:
860 case X_LP:
861#if BITS==64
862 buf_append(&s, &rem, "%d(%%rbp)", d * WORDSZ);
863#else
864 buf_append(&s, &rem, "%d(%%ebp)", d * WORDSZ);
865#endif
866 break;
867 case X_G:
868 case X_LG:
869#if BITS==64
870 buf_append(&s, &rem, "%d(%%rdi)", d * WORDSZ);
871#else
872 buf_append(&s, &rem, "%d(%%edi)", d * WORDSZ);
873#endif
874 break;
875 case X_L:
876 case X_LL:
877 buf_append(&s, &rem, "%s", label(d));
878 break;
879 case X_N:
880 buf_append(&s, &rem, "%d", d);
881 break;
882 }
883 }
884 va_end(ap);
885 emit("%s", buf);
886}

965static int rdn(void)
966{
967 bool neg;
968 int n;
969
970 neg = false;
971 n = 0;
972 do
973 {
974 ch = getchar();
975 } while (ch == ' ' || ch == '\n');
976 if (ch == 'L') {
977 ch = getchar();
978 } else if (ch == '-') {
979 neg = true;
980 ch = getchar();
981 }
982 while (isdigit(ch))
983 {
984 n = n * 10 + ch - '0';
985 ch = getchar();
986 }
987 return neg ? -n : n;
988}
